NODE = ELEMENT | STRUCTURAL_DIRECTIVE

    (@else-if(INTERPOLATION) STRUCTURAL_DIRECTIVE_BODY)*
    (@else) STRUCTURAL_DIRECTIVE_BODY

    (@case(string | number)STRUCTURAL_DIRECTIVE_BODY)*
    @default STRUCTURAL_DIRECTIVE_BODY
}
